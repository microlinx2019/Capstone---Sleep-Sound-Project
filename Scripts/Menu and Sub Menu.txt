GamePlayMainMenu:

using UnityEngine;
using System.Collections;
using Tkt.DungeonMaster.Actions;
using Tkt.DungeonMaster;
 
public class GameplayMainMenu : MonoBehaviour
{
    public GameObject MainMenuParent;
 
    void Awake()
    {
        HideMainMenu();
    }
 
    public void ToggleMainMenu()
    {
        MainMenuParent.SetActive(!MainMenuParent.activeSelf);
    }
 
    public void ShowMainMenu()
    {
        MainMenuParent.SetActive(true);
    }
 
    public void HideMainMenu()
    {
        MainMenuParent.SetActive(false);
    }
 
    public void GoToTown()
    {
        Command townCommand = new GoToTownCommand();
        var localPlayer = FindObjectOfType<LocalPlayerInputProcessor>();
        localPlayer.SetCommand(townCommand, true);
    }
}
